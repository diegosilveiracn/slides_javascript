\section{Introdução}

\begin{frame}
	\frametitle{JavaScript}
	
	\begin{itemize}
		\item É a linguagem de programação que dá comportamento as páginas web;
		\item Foi criado por Brendan Eich em 1995;
		\item Possui características multiparadigma e dinâmica; e
		\item Funciona no lado do cliente (diretamente no navegador).
	\end{itemize}
\end{frame}

\begin{frame}
\frametitle{Declaração dos scripts}

\begin{itemize}
	\item O código JavaScript pode ser declarado de três formas: 
	\begin{enumerate}
		\item Inline;
		\item Incorporado; ou
		\item Externo.
	\end{enumerate}
\end{itemize}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Script Inline}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
<input type="button" value="Clique" 
onclick="window.alert('Oi, mundo!');" >
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Script incorporado}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
<script>
	window.alert('Oi, mundo!');
</script>
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Script externo}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
window.alert('Oi, mundo!');			
		\end{lstlisting}
		
		\begin{lstlisting}
<script src="script.js"></script>
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\section{Entrada}

\begin{frame}[fragile]
	\frametitle{Variável e operador de atribuição}
	
	\begin{itemize}
		\item A declaração de variável pode acontecer de duas formas:
		\begin{enumerate}
			\item let; e
			\item const.
		\end{enumerate}
		\item Atribuição utiliza o símbolo \structure{=}.
	\end{itemize} \vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let a, b;
const  media = 60;
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Tipos de dados}

			\item Boolean;

			\item Undefined;
			\item BigInt;
			\item Symbol;

\begin{frame}[fragile]
	\frametitle{Operadores aritméticos}
	
	\begin{itemize}
		\item Adição: \structure{+};
		\item Subtração: \structure{-};
		\item Multiplicação: \structure{*}; 
		\item Divisão: \structure{/};
		\item Módulo: \structure{\%};
		\item Exponenciação:  \structure{**};
		\item Incremento: \structure{++}; e
		\item Decremento: \structure{- -}.
	\end{itemize}\vfill
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let a = 3, b = 5, soma;
soma = a + b;
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\section{Decisão}

\begin{frame}
	\frametitle{Estruturas de decisão}
	
	\begin{itemize}
		\item if/else; e
		\item switch.
	\end{itemize}	
\end{frame}

\begin{frame}[fragile]
	\frametitle{IF}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let numero = 4;
if (numero >= 0) {
	document.write('O número é positivo');
}else{
   document.write('O número é negativo');
}
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Switch}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let situacao = 1;
switch(sexo):
   case 0:
      document.write('Reprovado');
      break;
   case 1:
      document.write('Aprovado');
      break;
   default:
      document.write('Situação inválida');
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\section{Repetição}

\begin{frame}
	\frametitle{Estruturas de decisão}
	
	\begin{itemize}
		\item for;
		\item while;
		\item do/while; e
		\item forEach.
	\end{itemize}	
\end{frame}

\begin{frame}[fragile]
	\frametitle{For}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
for(let i = 0; i < 5; i++){
   document.write(i);
}
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{While}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let i = 0;
while(i < 5){
   document.write(i);
   i++;
}
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{Do/While}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
let i = 0;
do{
   document.write(i);
   i++;
}while (i < 5);
		\end{lstlisting}
	\end{exampleblock}
\end{frame}

\begin{frame}[fragile]
	\frametitle{forEach}
	
	\begin{exampleblock}{Exemplo}
		\begin{lstlisting}
frutas = ['maçã','banana','laranja'];
frutas.forEach((fruta, index) => {
   document.write('${index}:${fruta}');
});
		\end{lstlisting}
	\end{exampleblock}
\end{frame}
